-- Tabela principal para armazenar os resumos processados

CREATE TABLE RESUMOS_DOU (
    ID_RESUMO NUMBER GENERATED BY DEFAULT ON NULL AS IDENTITY PRIMARY KEY,
    ORGAO VARCHAR2(200),
    DATA_PUBLICACAO DATE,
    TITULO VARCHAR2(500),
    CONTEUDO CLOB,
    RESUMO CLOB
);

-- Tabela de controle para mensagens em dias sem publicação
CREATE TABLE AVISOS_DOU (
    ID_AVISO NUMBER GENERATED BY DEFAULT ON NULL AS IDENTITY PRIMARY KEY,
    DATA_REFERENCIA DATE,
    MENSAGEM VARCHAR2(500)
);
